Attribute VB_Name = "M13_PalabrasReservadas"
Option Explicit

'-------------------------------------------------------
'  M13_PalabrasReservadas
'  - Catálogo central de palabras reservadas, funciones,
'    operadores y elementos especiales de:
'      * VBA
'      * SQL_JET
'      * SQL_ACE
'      * ACCESS
'      * DAO
'      * ADO
'      * VB6
'      * OPERADORES
'      * FUNCIONES_VBA
'      * CONTROL_FLUJO_SISTEMA
'  - Soporta:
'      * múltiples categorías por palabra
'      * múltiples tipos por palabra
'      * búsqueda por categoría
'      * búsqueda por nombre (devuelve todas las coincidencias)
'-------------------------------------------------------

Public Enum ReservedCategory
    rcVBA = 1
    rcSQL_JET
    rcSQL_ACE
    rcACCESS
    rcDAO
    rcADO
    rcVB6
    rcOPERADORES
    rcFUNCIONES_VBA
    rcCONTROL_FLUJO_SISTEMA
End Enum

Public Type ReservedWordInfo
    Nombre As String
    Tipo As String
    Categoria As ReservedCategory
End Type

Private m_AllWords As Collection
Private m_Initialized As Boolean

'-------------------------------------------------------
'  API pública
'-------------------------------------------------------

Public Function GetAllReservedWords() As Collection
    EnsureInitialized
    Set GetAllReservedWords = m_AllWords
End Function

Public Function GetReservedWordsByCategory(ByVal cat As ReservedCategory) As Collection
    Dim col As New Collection
    Dim item As Variant
    Dim info As ReservedWordInfo

    EnsureInitialized

    For Each item In m_AllWords
        info = item
        If info.Categoria = cat Then
            col.Add info
        End If
    Next item

    Set GetReservedWordsByCategory = col
End Function

Public Function FindReservedWord(ByVal nombre As String) As Collection
    Dim col As New Collection
    Dim item As Variant
    Dim info As ReservedWordInfo
    Dim sName As String

    EnsureInitialized
    sName = LCase$(Trim$(nombre))

    For Each item In m_AllWords
        info = item
        If LCase$(info.Nombre) = sName Then
            col.Add info
        End If
    Next item

    Set FindReservedWord = col
End Function

'-------------------------------------------------------
'  Inicialización
'-------------------------------------------------------

Private Sub EnsureInitialized()
    If m_Initialized Then Exit Sub

    Set m_AllWords = New Collection

    Load_VBA
    Load_SQL_JET
    Load_SQL_ACE
    Load_ACCESS
    Load_DAO
    Load_ADO
    Load_VB6
    Load_OPERADORES
    Load_FUNCIONES_VBA
    Load_CONTROL_FLUJO_SISTEMA

    m_Initialized = True
End Sub

Private Sub AddWord( _
    ByVal nombre As String, _
    ByVal tipo As String, _
    ByVal cat As ReservedCategory _
)
    Dim info As ReservedWordInfo
    info.Nombre = nombre
    info.Tipo = tipo
    info.Categoria = cat
    m_AllWords.Add info
End Sub
